using System;
class HelloWorld {
  static void Main() {
      int n = int.Parse(Console.ReadLine());
      for (int i = 1;i<=n;i++){
          int sum=0;
          String a ="";
          for (int j =1;j<i+1;j++){
              if(i%j==0){
              a+='+';
              sum++;
              }
          }
          Console.WriteLine($"{i}{a}");
      }
  }
}
