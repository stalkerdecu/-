using System;
					
public class Program
{
	public static void Main()
	{
		int n = int.Parse(Console.ReadLine());
		int[] array = new int[n];
		int sum=0;
		for (int i=0;i<n;i++){
			array[i]=int.Parse(Console.ReadLine());
			sum+=array[i];
		}
		double x = sum/n;
		int q=0;
		int sumq=0;
		for (int i =0;i<n;i++){
			if(array[i]>x){
			q++;
			sumq+=array[i];
			}
			Console.WriteLine($"{q}");
		}
	}
}
